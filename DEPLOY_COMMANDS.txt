# FIFA WORLD CUP 2026 - CLOUD RUN DEPLOYMENT COMMANDS
# Copy and paste these commands one by one

# =============================================================================
# STEP 1: PREREQUISITES
# =============================================================================

# Check if gcloud is installed
gcloud --version

# If not installed, install it:
curl https://sdk.cloud.google.com | bash
exec -l $SHELL

# Login to Google Cloud
gcloud auth login

# Set your project ID (replace YOUR_PROJECT_ID with actual project ID)
gcloud config set project YOUR_PROJECT_ID

# Enable required APIs
gcloud services enable run.googleapis.com
gcloud services enable cloudbuild.googleapis.com
gcloud services enable containerregistry.googleapis.com

# =============================================================================
# STEP 2: PREPARE EXTERNAL SERVICES
# =============================================================================

# Get these ready before deployment:
# 1. MongoDB Atlas URI: https://www.mongodb.com/cloud/atlas
#    Format: mongodb+srv://username:password@cluster.mongodb.net/fifa-wc
#
# 2. OpenAI API Key: https://platform.openai.com/api-keys
#    Format: sk-...
#
# 3. Generate JWT Secret (at least 32 characters):
#    Run: openssl rand -base64 32

# =============================================================================
# STEP 3: DEPLOY TO CLOUD RUN (Choose ONE method)
# =============================================================================

# -----------------------------------------------------------------------------
# METHOD A: Using Deployment Script (EASIEST - RECOMMENDED)
# -----------------------------------------------------------------------------

chmod +x deploy-cloud-run.sh
./deploy-cloud-run.sh

# The script will prompt you for:
# - MongoDB URI
# - Redis URL (optional)
# - OpenAI API Key
# - JWT Secret

# -----------------------------------------------------------------------------
# METHOD B: Manual One-Line Deploy
# -----------------------------------------------------------------------------

# Replace the placeholder values with your actual values:
gcloud run deploy fifa-wc-assistant \
  --source . \
  --platform managed \
  --region us-central1 \
  --allow-unauthenticated \
  --memory 2Gi \
  --cpu 2 \
  --min-instances 0 \
  --max-instances 100 \
  --timeout 300 \
  --set-env-vars "NODE_ENV=production,PORT=8080,MONGODB_URI=YOUR_MONGODB_URI,OPENAI_API_KEY=YOUR_OPENAI_KEY,JWT_SECRET=YOUR_JWT_SECRET,DEFAULT_MODEL=gpt-4-turbo-preview,MAX_TOKENS=1000,TEMPERATURE=0.7,JWT_EXPIRE=7d"

# -----------------------------------------------------------------------------
# METHOD C: Using Docker (Advanced)
# -----------------------------------------------------------------------------

# Get your project ID
PROJECT_ID=$(gcloud config get-value project)

# Build Docker image
docker build -t gcr.io/$PROJECT_ID/fifa-wc-assistant:latest .

# Authenticate Docker with GCR
gcloud auth configure-docker

# Push to Google Container Registry
docker push gcr.io/$PROJECT_ID/fifa-wc-assistant:latest

# Deploy from container image
gcloud run deploy fifa-wc-assistant \
  --image gcr.io/$PROJECT_ID/fifa-wc-assistant:latest \
  --platform managed \
  --region us-central1 \
  --allow-unauthenticated \
  --memory 2Gi \
  --cpu 2 \
  --min-instances 0 \
  --max-instances 100 \
  --timeout 300

# Then set environment variables
gcloud run services update fifa-wc-assistant \
  --region us-central1 \
  --update-env-vars "NODE_ENV=production" \
  --update-env-vars "PORT=8080" \
  --update-env-vars "MONGODB_URI=YOUR_MONGODB_URI" \
  --update-env-vars "OPENAI_API_KEY=YOUR_OPENAI_KEY" \
  --update-env-vars "JWT_SECRET=YOUR_JWT_SECRET" \
  --update-env-vars "DEFAULT_MODEL=gpt-4-turbo-preview"

# =============================================================================
# STEP 4: VERIFY DEPLOYMENT
# =============================================================================

# Get your service URL
gcloud run services describe fifa-wc-assistant \
  --region us-central1 \
  --format 'value(status.url)'

# Test health endpoint (replace URL with your actual URL)
curl https://fifa-wc-assistant-xxxxx-uc.a.run.app/health

# View logs
gcloud run services logs tail fifa-wc-assistant --region us-central1

# =============================================================================
# TROUBLESHOOTING COMMANDS
# =============================================================================

# If deployment fails, check build logs
gcloud builds list --limit 5

# Get detailed build log (replace BUILD_ID with actual ID from above)
gcloud builds log BUILD_ID

# View service details
gcloud run services describe fifa-wc-assistant --region us-central1

# Test Docker build locally before deploying
docker build -t test-build .

# Test Docker container locally
docker run -p 8080:8080 \
  -e NODE_ENV=production \
  -e PORT=8080 \
  -e MONGODB_URI=your-uri \
  -e OPENAI_API_KEY=your-key \
  -e JWT_SECRET=your-secret \
  test-build
